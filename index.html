// 🔍 升級版：電話後三碼安全驗證（既簡單又不會改到別人的單） function searchMyOrder() { const inputPhone = document.getElementById('query-phone').value.trim(); const resultArea = document.getElementById('search-result'); if (!inputPhone) { alert('請輸入您填單時的聯絡電話以供查詢！'); return; } if (inputPhone.length < 3) { alert('請輸入正確且完整的電話號碼！'); return; } // 🔒 彈出安全核對視窗，要求輸入後三碼 const lastThreeDigits = prompt("【安全身分核對】\n為了防範誤改他人訂單，請輸入此號碼的「後三碼數字」："); if (!lastThreeDigits || lastThreeDigits.trim() === "") { alert("必須輸入正確的後三碼才能進行查詢與修改喔！"); resultArea.style.display = 'none'; return; } const targetCode = lastThreeDigits.trim(); // 取出使用者在查詢框輸入的電話的最後三碼，先跟使用者自己打的比對，做第一線防呆 const inputPhonePure = inputPhone.replace(/[^0-9]/g, ''); const inputLastThree = inputPhonePure.slice(-3); if (targetCode !== inputLastThree) { alert("❌ 驗證失敗：您輸入的後三碼與查詢的電話號碼不符！"); resultArea.style.display = 'none'; return; } resultArea.style.display = 'block'; resultArea.innerHTML = '🔍 正在進行安全身份核對中...'; fetch(`${FIREBASE_URL}/orders.json`) .then(res => res.json()) .then(data => { if (!data) { resultArea.innerHTML = '📭 目前雲端資料庫內尚無任何訂單紀錄。'; return; } globalOrdersCache = data; let html = `
✨ 身分核對成功！幫您找到以下填單紀錄：
`; let count = 0; for (let id in data) { let ord = data[id]; // 唯有資料庫內存的電話，與輸入的電話「完全一致」，才放行 const dbPhonePure = ord.phone ? ord.phone.replace(/[^0-9]/g, '') : ''; if (dbPhonePure === inputPhonePure) { count++; const statusColor = ord.status === '已確認' ? 'var(--success)' : 'red'; let editBtnHtml = ''; if (ord.status === '已確認') { editBtnHtml = `
💡 此訂單主揪已核對確認，如需修改請直接私訊主揪。`; } else { editBtnHtml = `
`; } html += `
🟢 姓名：${ord.name}
🍞 明細：${ord.items}
💰 金額：NT$ ${ord.total} 元 (${ord.payment})
📌 對帳狀態：${ord.status || '未核對'} ${editBtnHtml}
`; } } if (count === 0) { resultArea.innerHTML = `❌ 找不到與此電話號碼完全符合的訂單，請確認號碼是否輸入正確。`; } else { resultArea.innerHTML = html; } }) .catch(() => { resultArea.innerHTML = '❌ 查詢時連線出錯，請稍後再試。'; }); }
